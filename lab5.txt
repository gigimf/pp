int main()
{
  int vet[5][5];
  int i, j, k, val, exi;
  for (int i = 0; i < 5; i++){
    for (int j = 0; j < 5; j++){
      do{
        val = rand() % 100;     
        exi = 0;   
        for (int k = 0; k < j; k++){
          if (vet[i][k] == val){
            exi = 1;
            break;
          }
        }
        for (int k = 0; k < i; k++){
          if (vet[k][j] == val){
            exi = 1;
            break;
          }
        }
      }while (exi);
      vet[i][j] = val;
    }
  }
  for (i = 0; i < 5; i++){
    for (j = 0; j < 5; j++){
      printf("%d ", vet[i][j]);
    }
    printf("\n");
  }
  return 0;
}
